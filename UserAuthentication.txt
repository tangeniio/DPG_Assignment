CREATE PROCEDURE AuthenticateUser
	@Authentication_ID INT,
	@Authentication_Password VARCHAR(255),
	@Staff_ID INT,
	@Role VARCHAR(25) OUTPUT

AS
	SET NOCOUNT ON

BEGIN 	DECLARE @Username INT,
		DECLARE @Password VARCHAR(25),
		
		SELECT @Auth_ID =@Username, @Auth_Password =@Password
		
		IF(@Auth_ID = @Username COLLATE 250A)
			BEGIN
				SET @Role = "INCORRECT UserID"	
			END
		
		ELSE(@Auth_ID =@Password COLLATE 250A)
			BEGIN
				SET @Role = "INCORRECT PASSWORD"
			END
			
		ELSE 
			BEGIN 
				SET @Role = "LOGGED IN SUCCESSFULLY"
			END
			
	SELECT @Role
END
